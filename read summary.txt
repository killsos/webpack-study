entry：项目入口文件
output:构建的输出结果描述
  path 输出目录
  filename 输出文件名
  publicPath 输出目录所对应的外部路径 (从浏览器中访问)


  var path = require('path');
  module.exports = {
    entry:path.join(__dirname,'index'),
    output:{
      path:__dirname,
      filename:bundle.js
    },
    module:{
      loaders:[
        {
          test:"/\.css/"
          loaders:["style","css"]
        }
      ]
    }
  };
